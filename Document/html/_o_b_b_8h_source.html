231 // [ 1次元線分と点のはみ出し部分のベクトルを求める関数 ]
232 //
233 //++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
234 
235 inline void OverhangingPartOBB_Segment1D_Point( Vector3D* shortest_distance_vector,
236  const Vector3D* obb_position,
237  const Vector3D* obb_vector_half,
238  const Vector3D* obb_axis_vector,
239  const Vector3D* point )
240 {
241  // 長さが0以上か
242  float temp_length = obb_vector_half -> GetLength();
243  if ( temp_length <= 0.0f ) return;
244 
245  // 点から直線に垂線を降ろしたときの交点の位置を表す値(tとする)を算出
246  Vector3D temp_vector = *point - *obb_position;
247  float t = temp_vector.CreateVectorDot( *obb_axis_vector ) / temp_length;
248 
249  // はみ出した部分のベクトルを算出
250  if ( t < -1.0f )
251  {
252  *shortest_distance_vector += ( t + 1.0f ) * ( *obb_vector_half );
253  }
254  else if ( t > 1.0f )
255  {
256  *shortest_distance_vector += ( t - 1.0f ) * ( *obb_vector_half );
257  }
258 }
