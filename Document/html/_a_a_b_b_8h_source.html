102 inline float ShortestDistanceSquareAABB_Segment1D_Point(const float* position_min, const float* position_max, const float* point_any)
103 {
104  float length = 0.0f;
105 
106  // 線分より小さいとき
107  if (*point_any < *position_min)
108  {
109  length += (*position_min - *point_any) * (*position_min - *point_any);
110  }
111 
112  // 線分より大きいとき
113  if (*point_any > *position_max)
114  {
115  length += (*position_max - *point_any) * (*position_max - *point_any);
116  }
117 
118  return length;
119 }

129 inline float ShortestDistance_AABB_Point(const AABB* aabb, const Vector3D* point)

136  length = ShortestDistanceSquareAABB_Segment1D_Point(&aabb->min_.y, &aabb->max_.y, &point->y);
137  length = ShortestDistanceSquareAABB_Segment1D_Point(&aabb->min_.z, &aabb->max_.z, &point->z);
138 
139  return (float)sqrt(length);
